57 // Let's overload the stream input operator to read a list of CSV fields (which a CSV record).
58 // Remember, a record is a list of doubles separated by commas ','.
59 
60 istream& operator>>(istream& ins, record_t& record) {
61  // make sure that the returned record contains only the stuff we read now
62  record.clear();
63 
64  // read the entire line into a string (a CSV record is terminated by a newline)
65  string line;
66  getline(ins, line);
67 
68  // now we'll use a stringstream to separate the fields out of the line
69  stringstream ss(line);
70  string field;
71  while (getline(ss, field, ',')) {
72  // for each field we wish to convert it to a double
73  // (since we require that the CSV contains nothing but floating-point values)
74  stringstream fs(field);
75  double f = 0.0; // (default value is 0.0)
76  fs >> f;
77 
78  // add the newly-converted field to the end of the record
79  record.push_back(f);
80  }
81 
82  // Now we have read a single line, converted into a list of fields, converted the fields
83  // from strings to doubles, and stored the results in the argument record, so
84  // we just return the argument stream as required for this kind of input overload function.
85  return ins;
86 }

93 istream& operator>>(istream& ins, data_t& data) {
94  // make sure that the returned data only contains the CSV data we read here
95  data.clear();
96 
97  // For every record we can read from the file, append it to our resulting data
98  record_t record;
99  while (ins >> record) {
100  data.push_back(record);
101  }
102 
103  // Again, return the argument stream as required for this kind of input stream overload.
104  return ins;
